002/**
003 * This class is part of the "World of Zuul" application. 
004 * "World of Zuul" is a very simple, text based adventure game.  
005 * 
006 * This class holds an enumeration table of all command words known to the game.
007 * It is used to recognise commands as they are typed in.
008 *
009 * @author  Michael Kolling and David J. Barnes + D.Bureau
010 * @version 2008.03.30 + 2019.09.25
011 */
012public class CommandWords
013{
014    // a constant array that will hold all valid command words
015    private static final String[] aValidCommands = 
016    {"aller","quitter","aide","regarder","insÃ©rer","retour","test","valider","prendre","poser","inventaire","donner"};
017
018    /**
019     * Constructor - initialise the command words.
020     */
021    public CommandWords()
022    {
023        String[] vValidCommands = new String[12];
024        vValidCommands[0] = "aller";
025        vValidCommands[1] = "aide";
026        vValidCommands[2] = "quitter";
027        vValidCommands[3] = "regarder";
028        vValidCommands[4] = "insÃ©rer";
029        vValidCommands[5] = "retour";
030        vValidCommands[6] = "test";
031        vValidCommands[7] = "valider";
032        vValidCommands[8] = "prendre";
033        vValidCommands[9] = "poser";
034        vValidCommands[10]= "inventaire";
035        vValidCommands[11]= "donner";
036    } // CommandWords()
037
038    /**
039     * Check whether a given String is a valid command word. 
040     * @return true if a given string is a valid command,
041     * false if it isn't.
042     * @param pString qui prend l'input
043     */
044    public boolean isCommand(String pString )
045    {
046        for ( int i=0; i<this.aValidCommands.length; i++ ) {
047            if ( this.aValidCommands[i].equals( pString ) )
048                return true;
049        } // for
050        // if we get here, the string was not found in the commands
051        return false;
052    } // isCommand()
053    
054    
055    public String getCommandList ()
056    {
057        String vS = "";
058        for (String vCommand: aValidCommands){
059            vS += vCommand + " ";
060        }
061        return vS;
062       
063    }
064    
065    
066} // CommandWords
